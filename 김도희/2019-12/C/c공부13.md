1. 1 이상 100 미만의 정수 중에서 7의 배수와 9의 배수를 출력하는 프로그램을 작성하시오.

( 단 7 과 9 의 공배수는 1번만 출력할것 )

 

#include <stdio.h>

int main()
{
	int i;
	
	for(i=1;i<100;i++)
	{
		if((i%7==0 || i%9==0))
		{
			printf("%d ", i);
		}
	}
	
	return 0;
}

문제 8-2
 

 

1. 구구단을 출력하되 짝수 단만 출력하시오 ( 단 2단은 2 x 2 , 4단은 4 x 4...과 같은 방식으로 출력할 것 )

 

#include <stdio.h>

int main()
{
	int i=0,j=1;
	
	while(i<9)
	{
		for(j=1;j<=i;j++)
		{
			if(j==i)
			{
				printf("%d x %d = %2d",i,j,i*j);
				break;
			}
			
			else
			printf("%d x %d = %2d\n",i,j,i*j);
		}
		
		printf("\n\n");
		
		i+=2;
	}
	
	return 0;
}
