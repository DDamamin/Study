# 똥멍청이(멘토선배)에 대한 정보

####  구조체를 사용

```c
#include <stdio.h>
#include <string.h>

struct Senior{
	char name[100];
	int age;
	int birday;
	char nickname[10];
};

int main(){
	struct Senior s1;
	strcpy(s1.name, "똥멍청이");
	s1.age = 18;
	s1.birday = 915;
	strcpy(s1.nickname, "유시온");
	printf("이름 : %s", s1.name);
	printf("나이 : %d", s1.age);
	printf("생일 : %d", s1.birday);
	printf("별명 : %s", s1.nickname);
	return 0;
}
```
