# 업 남선배님들 별명

**구조체(배열)를 사용**

[2020.08.02.일]

```c
#include <stdio.h>
#include <string.h>

typedef struct Senior{
	char name[5];
	int age;
	char nickname[100];
}_up; 

int main(){
	_up up[6];
	int i; 
	strcpy(up[0].name, "이진혁");
	up[0].age = 18;
	strcpy(up[0].nickname, "오늘 죽는 사람");
	
	strcpy(up[1].name, "손정우");
	up[1].age = 18;
	strcpy(up[1].nickname, "부장과 종은이와 친해지길 바라는 사람");
	
	strcpy(up[2].name, "서민준");
	up[2].age = 18;
	strcpy(up[2].nickname, "내 학동 선생님");
	
	strcpy(up[3].name, "손완서");
	up[3].age = 18;
	strcpy(up[3].nickname, "놀리고 싶은 사람");
	
	strcpy(up[4].name, "안영준");
	up[4].age = 18;
	strcpy(up[4].nickname, "귀여운 곰돌이");
	
	strcpy(up[5].name, "유시온");
	up[5].age = 18;
	strcpy(up[5].nickname, "펭이 뺏어간 사람");
	
	for(i=0;i<6;i++){
		printf("%s\n", up[i].name);
		printf("%d\n", up[i].age);
		printf("%s\n", up[i].nickname);
	}
	return 0;
}
```
